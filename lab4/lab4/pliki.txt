#include <stdio.h>
#include <string.h>

#define MAX_STR_LEN 10
#define NUM_STRINGS 5


void concat(char strings[NUM_STRINGS][MAX_STR_LEN + 1], char* result, size_t result_size);

int main() {
    char strings[NUM_STRINGS][MAX_STR_LEN + 1]; 
    char result[(MAX_STR_LEN + 1) * NUM_STRINGS] = ""; 
    size_t result_size = sizeof(result); 
    int i;

    memset(strings, 0, sizeof(strings));

    printf("Wprowadz %d lancuchow znakow (maksymalnie %d znakow kazdy):\n", NUM_STRINGS, MAX_STR_LEN);

    for (i = 0; i < NUM_STRINGS; i++) {
        printf("Lancuch %d: ", i + 1);
        if (fgets(strings[i], sizeof(strings[i]), stdin) == NULL) {
            fprintf(stderr, "Blad podczas wczytywania danych.\n");
            return 1;
        }

        size_t len = strlen(strings[i]);
        if (len > 0 && strings[i][len - 1] == '\n') {
            strings[i][len - 1] = '\0';
        }
    }

    concat(strings, result, result_size);

    printf("\nWynikowy tekst: %s\n", result);

    return 0;
}
